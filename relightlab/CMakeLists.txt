	relightlab PUBLIC 
		${CMAKE_CURRENT_SOURCE_DIR}
		${JPEG_INCLUDE_DIR}
		TIFF::TIFF
		${EIGEN3_INCLUDE_DIR}
		../external
	)

	relightlab PUBLIC

		TIFF::TIFF
		OpenMP::OpenMP_CXX
		opencv_core  opencv_imgcodecs opencv_imgproc opencv_video

		${RELIGHT_QT}::Xml
